Módulo Anti-Fraude - Detalhes Minuciosos
O que é?
Sistema de detecção e prevenção de fraudes no setor de telecom. Monitora quando seus clientes inadimplentes são consultados por outros provedores, indicando possível tentativa de "golpe" (cancelar devendo e contratar em outro lugar).

🎯 PROBLEMA QUE RESOLVE
O Golpe Clássico do Setor
1️⃣ Cliente contrata Provedor A
         ↓
2️⃣ Recebe equipamento em comodato (ONU, roteador)
         ↓
3️⃣ Para de pagar após alguns meses
         ↓
4️⃣ Ignora cobranças do Provedor A
         ↓
5️⃣ Contrata Provedor B (concorrente)
         ↓
6️⃣ Provedor A fica com:
   • Faturas em aberto
   • Equipamento não devolvido
   • Prejuízo financeiro
Com o Anti-Fraude
1️⃣ Cliente do Provedor A está inadimplente
         ↓
2️⃣ Provedor B consulta CPF do cliente
         ↓
3️⃣ Sistema DETECTA a consulta
         ↓
4️⃣ Provedor A recebe ALERTA IMEDIATO:
   "⚠️ Seu cliente João foi consultado pela Empresa B!
    Pendências: R$ 450 + 1 ONU não devolvida"
         ↓
5️⃣ Provedor A pode:
   • Intensificar cobrança
   • Agendar retirada do equipamento
   • Negativar no SPC
   • Acionar juridicamente

// Quando Provedor B consulta um CPF:

async function performIspConsultation(document, consultingProviderId) {
  
  // 1. Busca cliente em todos os provedores
  const customers = await Customer.filter({ document });
  
  for (const customer of customers) {
    
    // 2. Pula se é do próprio provedor
    if (customer.provider_id === consultingProviderId) continue;
    
    // 3. Carrega configuração anti-fraude do provedor original
    const config = await AntiFraudConfig.filter({ 
      provider_id: customer.provider_id 
    });
    
    // 4. Verifica se deve gerar alerta
    const shouldAlert = evaluateAlertConditions(customer, config);
    
    if (shouldAlert) {
      // 5. Calcula score de risco
      const riskData = calculateRiskScore(customer);
      
      // 6. Cria o alerta
      await AntiFraudAlert.create({
        customer_id: customer.id,
        original_provider_id: customer.provider_id,
        consulting_provider_id: consultingProviderId,
        consulting_provider_name: consultingProvider.name,
        reason: buildAlertReason(customer),
        status: 'new',
        flagged_date: new Date().toISOString(),
        risk_score: riskData.score,
        risk_level: riskData.level,
        risk_factors: riskData.factors,
        days_overdue: customer.max_days_overdue,
        overdue_amount: customer.total_overdue_amount,
        // ... demais campos
      });
      
      // 7. Envia notificação por email (se configurado)
      if (config.notify_on_critical && riskData.level === 'critical') {
        await sendAlertEmail(config.notification_emails, alertData);
      }
    }
  }
}
📈 CÁLCULO DO SCORE DE RISCO
Fórmula Detalhada
function calculateRiskScore(customer, equipment, alerts) {
  let score = 0;
  let factors = [];
  
  // ═══════════════════════════════════════════════════════
  // FATOR 1: DIAS DE ATRASO (máx 35 pontos)
  // ═══════════════════════════════════════════════════════
  const daysOverdue = customer.max_days_overdue || 0;
  
  if (daysOverdue >= 90) {
    score += 35;
    factors.push("Atraso superior a 90 dias");
  } else if (daysOverdue >= 60) {
    score += 25;
    factors.push("Atraso entre 60-90 dias");
  } else if (daysOverdue >= 30) {
    score += 15;
    factors.push("Atraso entre 30-60 dias");
  } else if (daysOverdue >= 1) {
    score += 8;
    factors.push("Atraso de 1-30 dias");
  }
  
  // ═══════════════════════════════════════════════════════
  // FATOR 2: VALOR EM ABERTO (máx 25 pontos)
  // ═══════════════════════════════════════════════════════
  const overdueAmount = customer.total_overdue_amount || 0;
  
  if (overdueAmount >= 1000) {
    score += 25;
    factors.push(`Valor alto em aberto: R$ ${overdueAmount.toFixed(2)}`);
  } else if (overdueAmount >= 500) {
    score += 18;
    factors.push(`Valor médio em aberto: R$ ${overdueAmount.toFixed(2)}`);
  } else if (overdueAmount >= 200) {
    score += 10;
    factors.push(`Valor em aberto: R$ ${overdueAmount.toFixed(2)}`);
  } else if (overdueAmount > 0) {
    score += 5;
    factors.push(`Pequeno valor em aberto: R$ ${overdueAmount.toFixed(2)}`);
  }
  
  // ═══════════════════════════════════════════════════════
  // FATOR 3: EQUIPAMENTOS NÃO DEVOLVIDOS (máx 25 pontos)
  // ═══════════════════════════════════════════════════════
  const notReturnedEquipment = equipment.filter(e => 
    e.status === 'not_returned' || e.status === 'installed'
  );
  const equipmentValue = notReturnedEquipment.reduce(
    (sum, e) => sum + (e.value || 0), 0
  );
  
  if (equipmentValue >= 500) {
    score += 25;
    factors.push(`Equipamento de alto valor: R$ ${equipmentValue.toFixed(2)}`);
  } else if (equipmentValue >= 200) {
    score += 18;
    factors.push(`Equipamento não devolvido: R$ ${equipmentValue.toFixed(2)}`);
  } else if (notReturnedEquipment.length > 0) {
    score += 10;
    factors.push(`${notReturnedEquipment.length} equipamento(s) pendente(s)`);
  }
  
  // ═══════════════════════════════════════════════════════
  // FATOR 4: CONSULTAS RECENTES (máx 15 pontos)
  // ═══════════════════════════════════════════════════════
  const recentAlerts = alerts.filter(a => {
    const alertDate = new Date(a.flagged_date);
    const thirtyDaysAgo = new Date();
    thirtyDaysAgo.setDate(thirtyDaysAgo.getDate() - 30);
    return alertDate >= thirtyDaysAgo;
  });
  
  if (recentAlerts.length >= 5) {
    score += 15;
    factors.push(`Consultado por ${recentAlerts.length} provedores recentemente`);
  } else if (recentAlerts.length >= 3) {
    score += 10;
    factors.push(`Múltiplas consultas recentes: ${recentAlerts.length}`);
  } else if (recentAlerts.length >= 2) {
    score += 5;
    factors.push("Consultado por 2 provedores nos últimos 30 dias");
  }
  
  // ═══════════════════════════════════════════════════════
  // FATOR 5: IDADE DO CONTRATO (máx 10 pontos)
  // ═══════════════════════════════════════════════════════
  const contractStart = new Date(customer.contract_start_date);
  const contractAgeDays = Math.floor(
    (new Date() - contractStart) / (1000 * 60 * 60 * 24)
  );
  
  if (contractAgeDays < 30) {
    score += 10;
    factors.push("Contrato muito recente (< 30 dias)");
  } else if (contractAgeDays < 90) {
    score += 5;
    factors.push("Contrato recente (< 90 dias)");
  }
  
  // ═══════════════════════════════════════════════════════
  // DETERMINAÇÃO DO NÍVEL DE RISCO
  // ═══════════════════════════════════════════════════════
  let riskLevel;
  if (score >= 75) {
    riskLevel = 'critical';
  } else if (score >= 50) {
    riskLevel = 'high';
  } else if (score >= 25) {
    riskLevel = 'medium';
  } else {
    riskLevel = 'low';
  }
  
  return { score, level: riskLevel, factors };
}
